var Html = {
  load: function() {
    var s=document.getElementById("show_drug");
    if(typeof(s)=="undefined"||s==null)
        return arr["default"];
    var code = s.src.replace(/^.+\?/,'');
    code=decodeURI(code);
    if(typeof(arr[code])!="undefined")
        return arr[code];
    var index;
    while((index=code.lastIndexOf('_'))!=-1){
        code=code.substring(0,index);
        if(typeof(arr[code])!="undefined")
        {
            return arr[code];
                 }
    }
    return arr["default"];
  }
}
